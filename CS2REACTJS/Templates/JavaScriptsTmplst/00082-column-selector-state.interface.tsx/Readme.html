    interface AppState {
        colToSel: Array<IColumnSelectorItemInterface>
        open: boolean;
    }

    interface AppProps {}


    class App extends React.Component < AppProps, AppState > {
      constructor(props: AppProps) {
        super(props);
        this.state =  {
          colToSel:[
            {name: 'int32', caption: 'caption int32', checked: false },
            {name: 'double', caption: 'caption double', checked: true },
            {name: 'string', caption: 'caption string', checked: false },
            {name: 'datetime', caption: 'caption datetime', checked: true },

          ],
          open: false,

        };    
        this.onClick = this.onClick.bind(this);
        this.onClose = this.onClose.bind(this);
      }


      onClick() {
        this.setState<'open'>( { open: true } );
      }

      onClose(rslt: boolean ,cols?: Array < IColumnSelectorItemInterface >) {
        this.setState<'open'>( { open: false } );
        if(cols) {
          cols.forEach(c => {
            let  i: number = this.state.colToSel.findIndex(v=>v.name===c.name)
            if(i > -1) this.state.colToSel[i].checked = c.checked
          })
          this.setState<'colToSel'>( { colToSel: this.state.colToSel } );
        }
      }


      render() {
        return (
            <div>
                <div>
                <ColumnSelector columsToSelect={this.state.colToSel}></ColumnSelector>
                <p />

                <button onClick={this.onClick}>ColumnSelectorDlg</button> 
                <ColumnSelectorDlg open={ this.state.open }
                                   maxWidth='md'
                                   colums={this.state.colToSel}
                                   caption={'this is caption'}
                                   onClose={this.onClose}></ColumnSelectorDlg> 
              </div>
          </div>
        )};
