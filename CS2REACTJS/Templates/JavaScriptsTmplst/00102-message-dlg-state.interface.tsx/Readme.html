    interface AppState {
        openmsg: boolean;
    }

    interface AppProps {}


    class App extends React.Component {
      constructor(props: AppProps) {
        super(props);
        this.state =  {
            openmsg: false
        };    
        this.onClickMsg = this.onClickMsg.bind(this);
        this.onCloseMsg = this.onCloseMsg.bind(this);
      }


      onClickMsg() {
        this.setState<'openmsg'>( { openmsg: true } );
      }
      onCloseMsg(v: boolean) {
        this.setState<'openmsg'>( { openmsg: false } );
      }

      render() {
        return (
            <div>
                <div>
                  <MessageDlg message={'this is a message'} open={this.state.openmsg} onClose={this.onCloseMsg}></MessageDlg>
                  <p/>
                  <button onClick={this.onClickMsg} >MessageDlg</button>
              </div>
          </div>
        )};
